graph = eval(input("Enter your graph: "))
node = input("Enter your starting node: ")

visited = set()

def dfs(visited, graph, node):
    if node not in visited:
        print(node, end=" ")
        visited.add(node)

        for neighbour in graph[node]:
            dfs(visited, graph, neighbour)

print("DFS:")
dfs(visited, graph, node)
